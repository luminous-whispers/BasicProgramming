char * StrConstCpy(char *source, int len)
{
    char *ptr;
    int i;

    ptr = (char *)malloc(len);
    if (ptr == NULL){
      printf("Memory allocation failed!\n");
      exit(0);
    }
    for (i=0; i<len; i++)
        *(ptr+i) = *(source+i);
    return ptr;
}
