void bubbleSort(int a[],int n)
{
	int hold;
	for(int i=0;i<n-1;i++){
		for(int j=0;j<n-1-i;j++){
			if(a[j]>a[j+1])
			{
				hold=a[j];
				a[j]=a[j+1];
				a[j+1]=hold;
			}
		}
	}
	return ;
}

void bubbleSort2(int a[],int n)
{
	int hold;
	for(int i=0;i<n-1;i++){
		int flag=0;
		for(int j=0;j<n-1-i;j++){
			if(a[j]>a[j+1])
			{
				hold=a[j];
				a[j]=a[j+1];
				a[j+1]=hold;
				flag=1;
			}
		}
		if(flag==0){//j循环后有交换则已，没有交换就推出 
			break;
		}
	}
}
